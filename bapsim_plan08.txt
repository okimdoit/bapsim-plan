function getToday(){
  const d=new Date();
  const day=d.getDay(); // 0=일,1=월,...
  return day===0?6:day-1; // 0=월,...,6=일
}

function getWeekRange(){
  const now=new Date();
  const day=now.getDay();
  const mon=new Date(now); mon.setDate(now.getDate()-(day===0?6:day-1));
  const sun=new Date(mon); sun.setDate(mon.getDate()+6);
  const fmt=d=>`${d.getMonth()+1}월${d.getDate()}일`;
  const weekNum=getWeekNumber(now);
  return { week:`${weekNum}주차`, range:`${fmt(mon)}~${fmt(sun)}` };
}

function getLastWeekRange(){
  const now=new Date();
  now.setDate(now.getDate()-7);
  const day=now.getDay();
  const mon=new Date(now); mon.setDate(now.getDate()-(day===0?6:day-1));
  const sun=new Date(mon); sun.setDate(mon.getDate()+6);
  const fmt=d=>`${d.getMonth()+1}월 ${d.getDate()}일`;
  const weekNum=getWeekNumber(mon);
  return { week:`${weekNum}주차`, range:`${fmt(mon)} ~ ${fmt(sun)}` };
}

function getWeekNumber(d){
  const date=new Date(Date.UTC(d.getFullYear(),d.getMonth(),d.getDate()));
  date.setUTCDate(date.getUTCDate()+4-(date.getUTCDay()||7));
  const yearStart=new Date(Date.UTC(date.getUTCFullYear(),0,1));
  return Math.ceil((((date-yearStart)/86400000)+1)/7);
}

function applyTaskStates(){
  if(!window._taskStates) return;
  Object.keys(window._taskStates).forEach(key=>{
    const [day,id]=key.split('__');
    const row=document.querySelector(`#taskBody-${day} tr[data-id="${id}"]`);
    if(!row) return;
    const cb=row.querySelector('input[type=checkbox]');
    if(window._taskStates[key]){
      if(cb) cb.checked=true;
      moveCheckedToBottom(row,day);
    }
  });
  Object.keys(window._taskStars||{}).forEach(key=>{
    const [day,id]=key.split('__');
    const row=document.querySelector(`#taskBody-${day} tr[data-id="${id}"]`);
    if(!row) return;
    const stars=row.querySelector('.stars');
    if(stars) applyStarDisplay(stars, window._taskStars[key]||0);
  });
  for(let i=0;i<7;i++) sortTasksByStars(i);
}

function taskCheck(cb,day,id){
  const row=cb.closest('tr');
  const key=day+'__'+id;
  window._taskStates[key]=cb.checked;
  if(cb.checked) moveCheckedToBottom(row,day);
  else {
    const tbody=document.getElementById('taskBody-'+day);
    if(tbody) tbody.insertBefore(row,tbody.firstChild);
  }
  fbSet('bapsim/taskStates',window._taskStates);
}
window.taskCheck=taskCheck;

function moveCheckedToBottom(row,day){
  const tbody=document.getElementById('taskBody-'+day);
  if(tbody) tbody.appendChild(row);
}

function setStars(starsEl,day,id,e){
  if(e) e.stopPropagation();
  const starEls=starsEl.querySelectorAll('.star');
  const rect=starsEl.getBoundingClientRect();
  const x=(e?e.clientX:0)-rect.left;
  const w=rect.width/3;
  let n=Math.ceil(x/w);
  n=Math.max(0,Math.min(3,n));
  /* 같은 값 클릭하면 0으로 */
  const cur=window._taskStars[day+'__'+id]||0;
  if(cur===n) n=0;
  window._taskStars[day+'__'+id]=n;
  applyStarDisplay(starsEl,n);
  fbSet('bapsim/taskStars',window._taskStars);
  sortTasksByStars(parseInt(day));
}
window.setStars=(el,day,id)=>{ el.onclick=(e)=>setStars(el,day,id,e); };

/* stars onclick 직접 처리 — HTML에서 onclick="setStars(this,'0','m1')" 형태로 호출 */
window.setStars=function(el,day,id){
  el.onclick=function(e){
    e.stopPropagation();
    const starEls=el.querySelectorAll('.star');
    const rect=el.getBoundingClientRect();
    const x=e.clientX-rect.left;
    const w=rect.width/3;
    let n=Math.ceil(x/w);
    n=Math.max(0,Math.min(3,n));
    const cur=window._taskStars[day+'__'+id]||0;
    if(cur===n) n=0;
    window._taskStars[day+'__'+id]=n;
    applyStarDisplay(el,n);
    fbSet('bapsim/taskStars',window._taskStars);
    sortTasksByStars(parseInt(day));
  };
};

function applyStarDisplay(starsEl,n){
  starsEl.querySelectorAll('.star').forEach((s,i)=>{
    s.classList.toggle('on',i<n);
  });
}

function sortTasksByStars(day){
  const tbody=document.getElementById('taskBody-'+day);
  if(!tbody) return;
  const rows=[...tbody.querySelectorAll('tr')];
  const unchecked=rows.filter(r=>!r.querySelector('input[type=checkbox]')?.checked);
  const checked=rows.filter(r=>r.querySelector('input[type=checkbox]')?.checked);
  unchecked.sort((a,b)=>{
    const idA=a.dataset.id; const idB=b.dataset.id;
    const sA=window._taskStars[day+'__'+idA]||0;
    const sB=window._taskStars[day+'__'+idB]||0;
    return sB-sA;
  });
  [...unchecked,...checked].forEach(r=>tbody.appendChild(r));
}

function postponeTask(fromDay,id,toDay){
  const row=document.querySelector(`#taskBody-${fromDay} tr[data-id="${id}"]`);
  if(!row) return;
  const cells=row.querySelectorAll('td');
  const taskText=cells[2]?cells[2].textContent.trim():'';
  const time=cells[1]?cells[1].textContent.trim():'';
  const room=cells[3]?cells[3].textContent.trim():'';
  const dur=cells[4]?cells[4].textContent.trim():'';
  row.remove();
  /* 다음날 커스텀 태스크로 추가 */
  if(!window._customTasks[toDay]) window._customTasks[toDay]=[];
  const newId='p'+Date.now();
  window._customTasks[toDay].unshift({id:newId,text:`[이월] ${taskText}`,room,time,dur,stars:3});
  renderCustomTasks();
  fbSet('bapsim/customTasks',window._customTasks);
  delete window._taskStates[fromDay+'__'+id];
  fbSet('bapsim/taskStates',window._taskStates);
}
window.postponeTask=postponeTask;

function addTask(day){
  const inp=document.getElementById('newTask-'+day);
  if(!inp||!inp.value.trim()) return;
  if(!window._customTasks[day]) window._customTasks[day]=[];
  const id='ct'+Date.now();
  window._customTasks[day].unshift({id,text:inp.value.trim(),room:'',time:'',dur:'',stars:0});

  fbSet('bapsim/customTasks',window._customTasks);

window.addTask=addTask;

function renderCustomTasks(){
  for(let day=0;day<7;day++){
    const tasks=window._customTasks[day]||[];
    const existingCustom=document.querySelectorAll(`#taskBody-${day} tr[data-custom]`);
    existingCustom.forEach(r=>r.remove());
    const tbody=document.getElementById('taskBody-'+day);
    if(!tbody) continue;
    tasks.forEach(t=>{
      const tr=document.createElement('tr');
      tr.dataset.id=t.id;
      tr.dataset.custom='1';
      const isChecked=window._taskStates[day+'__'+t.id];
      if(isChecked) tr.classList.add('done-row');
      const stars=t.stars||0;
      const dayNames=['월','화','수','목','금','토','일'];
      const nextDay=(day+1)%7;
      tr.innerHTML=`
        <td><input type="checkbox" ${isChecked?'checked':''} onchange="taskCheck(this,'${day}','${t.id}')"></td>
        <td>${t.time||'—'}</td>
        <td contenteditable="true" onblur="updateCustomTask(${day},'${t.id}','text',this.textContent)">${t.text}</td>
        <td>${t.room||'—'}</td>
        <td>${t.dur||'—'}</td>
        <td><div class="stars" id="stars-${day}-${t.id}"><span class="star ${stars>=1?'on':''}">★</span><span class="star ${stars>=2?'on':''}">★</span><span class="star ${stars>=3?'on':''}">★</span></div></td>
        <td><button class="btn-postpone" onclick="postponeTask('${day}','${t.id}',${nextDay})">미루기→${dayNames[nextDay]}</button>
            <button class="btn-del" onclick="deleteCustomTask(${day},'${t.id}')" style="margin-left:4px;">삭제</button></td>
      `;
      /* 별점 클릭 */
      const starsEl=tr.querySelector('.stars');
      starsEl.onclick=function(e){
        e.stopPropagation();
        const rect=starsEl.getBoundingClientRect();
        const x=e.clientX-rect.left;
        const w=rect.width/3;
        let n=Math.ceil(x/w);
        n=Math.max(0,Math.min(3,n));
        const cur=window._taskStars[day+'__'+t.id]||0;
        if(cur===n) n=0;
        window._taskStars[day+'__'+t.id]=n;
        t.stars=n;
        applyStarDisplay(starsEl,n);
        fbSet('bapsim/taskStars',window._taskStars);
        sortTasksByStars(day);
      };
      if(isChecked) tbody.appendChild(tr);
      else tbody.insertBefore(tr,tbody.firstChild);
    });
    sortTasksByStars(day);
  }
}

function updateCustomTask(day,id,field,val){
  const tasks=window._customTasks[day]||[];
  const t=tasks.find(x=>x.id===id);
  if(t){ t[field]=val; fbSet('bapsim/customTasks',window._customTasks); }
}
window.updateCustomTask=updateCustomTask;

function deleteCustomTask(day,id){
  window._customTasks[day]=(window._customTasks[day]||[]).filter(t=>t.id!==id);
  document.querySelector(`#taskBody-${day} tr[data-id="${id}"]`)?.remove();
  fbSet('bapsim/customTasks',window._customTasks);
}
window.deleteCustomTask=deleteCustomTask;

/* 예약 콘텐츠 자동 완료 */
function autoCompleteScheduled(){
  const now=new Date();
  const todayDay=getToday();
  const rows=window._uploadRows[todayDay]||[];
  let changed=false;
  rows.forEach(r=>{
    if(r.status==='예약' && r.time){
      const [hh,mm]=(r.time+':00').split(':').map(Number);
      const scheduled=new Date(); scheduled.setHours(hh,mm,0,0);
      if(now>=scheduled){ r.status='완료'; changed=true; }
    }
  });
  if(changed){ renderUploadRows(todayDay); fbSet('bapsim/uploadRows',window._uploadRows); }
}

/* CSV 내보내기 */
function exportUploadCSV(day){
  const rows=window._uploadRows[day]||[];
  const dayNames=['월','화','수','목','금','토','일'];
  let csv='시간,플랫폼,콘텐츠 제목,상태,비고\n';
  rows.forEach(r=>{
    csv+=`"${r.time||''}","${r.platform||''}","${r.title||''}","${r.status||''}","${r.note||''}"\n`;
  });
  const blob=new Blob(['\uFEFF'+csv],{type:'text/csv;charset=utf-8'});
  const a=document.createElement('a');
  a.href=URL.createObjectURL(blob);
  a.download=`업로드일정_${dayNames[day]}요일.csv`;
  a.click();
}
window.exportUploadCSV=exportUploadCSV;

/* CSV 가져오기 */
function importUploadCSV(e,day){
  const file=e.target.files[0]; if(!file) return;
  const reader=new FileReader();
  reader.onload=ev=>{
    const lines=ev.target.result.split('\n').slice(1);
    if(!window._uploadRows[day]) window._uploadRows[day]=[];
    lines.forEach(line=>{
      if(!line.trim()) return;
      const cols=line.split(',').map(c=>c.replace(/^"|"$/g,'').trim());
      window._uploadRows[day].push({id:'csv'+Date.now()+Math.random(),time:cols[0],platform:cols[1],title:cols[2],status:cols[3],note:cols[4]});
    });
    renderUploadRows(day);
    fbSet('bapsim/uploadRows',window._uploadRows);
  };
  reader.readAsText(file,'UTF-8');
}
window.importUploadCSV=importUploadCSV;

function renderMsGroup(key){
  const cont=document.getElementById('ms-'+key);
  if(!cont) return;
  if(!window._msItems[key]){
    window._msItems[key]=JSON.parse(JSON.stringify(MS_DEFAULTS[key]||[]));
  }
  const items=window._msItems[key];
  cont.innerHTML='';
  items.forEach((item,idx)=>{
    const row=document.createElement('div');
    row.style.cssText='display:flex;align-items:center;gap:6px;padding:4px 0;border-bottom:1px dashed var(--border);';
    row.innerHTML=`
      <input type="checkbox" ${item.done?'checked':''} onchange="msCheck('${key}',${idx},this.checked)">
      <span contenteditable="true" style="flex:1;font-size:.82rem;${item.done?'text-decoration:line-through;opacity:.5':''}"
        onblur="msSaveText('${key}',${idx},this.textContent)">${item.text}</span>
      <div class="stars" style="cursor:pointer;" id="ms-stars-${key}-${idx}">
        ${[1,2,3].map(n=>`<span class="star ${(item.stars||0)>=n?'on':''}">★</span>`).join('')}
      </div>
      <button class="btn-del" onclick="msDel('${key}',${idx})">×</button>
    `;
    const starsEl=row.querySelector('.stars');
    starsEl.onclick=function(e){
      e.stopPropagation();
      const rect=starsEl.getBoundingClientRect();
      const x=e.clientX-rect.left;
      let n=Math.ceil(x/(rect.width/3));
      n=Math.max(0,Math.min(3,n));
      if((item.stars||0)===n) n=0;
      item.stars=n;
      applyStarDisplay(starsEl,n);
      fbSet('bapsim/msItems',window._msItems);
    };
    cont.appendChild(row);
  });
}

function renderGuideChecklist(roomIdx){
  const cont=document.getElementById('guideChecklist'+roomIdx);
  if(!cont) return;
  const items=window._guideChecks[roomIdx]||[];
  if(!items.length){ cont.innerHTML='<div style="color:var(--muted);font-size:.8rem;padding:4px 0;">항목 없음</div>'; return; }
  let html='<table class="cl-table"><thead><tr><th>✓</th><th>할 일</th><th>중요도</th><th>삭제</th></tr></thead><tbody>';
  items.forEach((item,idx)=>{
    html+=`<tr class="${item.done?'checked-row':''}">
      <td><input type="checkbox" ${item.done?'checked':''} onchange="guideCheckToggle(${roomIdx},${idx},this.checked)"></td>
      <td><span contenteditable="true" style="font-size:.82rem;" onblur="guideCheckEdit(${roomIdx},${idx},this.textContent)">${item.text}</span></td>

    </tr>`;
  });
  html+='</tbody></table>';
  cont.innerHTML=html;
  /* 별점 이벤트 */
  items.forEach((item,idx)=>{
    const el=document.getElementById(`gcs-${roomIdx}-${idx}`);
    if(!el) return;
    el.onclick=function(e){
      e.stopPropagation();

      n=Math.max(0,Math.min(3,n));

      item.stars=n;
      applyStarDisplay(el,n);
      fbSet('bapsim/guideChecks',window._guideChecks);
    };
  });
}

function guideCheckToggle(roomIdx,idx,done){
  if(!window._guideChecks[roomIdx]) return;
  window._guideChecks[roomIdx][idx].done=done;
  /* 오늘 할일에 자동 반영 */
  const today=getToday();
  const text=window._guideChecks[roomIdx][idx].text;
  if(done){
    if(!window._customTasks[today]) window._customTasks[today]=[];
    const exists=window._customTasks[today].find(t=>t.text===`[지침] ${text}`);
    if(!exists){
      window._customTasks[today].unshift({id:'gc'+Date.now(),text:`[지침] ${text}`,room:'AI지침',time:'',dur:'',stars:window._guideChecks[roomIdx][idx].stars||0});
      renderCustomTasks();
      fbSet('bapsim/customTasks',window._customTasks);
    }
  }
  renderGuideChecklist(roomIdx);
  fbSet('bapsim/guideChecks',window._guideChecks);
}
window.guideCheckToggle=guideCheckToggle;

  const el=document.getElementById('floatMemoText');

let fmAutoTimer;

  const el=document.getElementById('floatMemoText');
  if(el) el.addEventListener('input',()=>{

    fmAutoTimer=setTimeout(saveFloatMemo,2000);
  });

function initFloatMemoDrag(){
  const wrap=document.getElementById('floatMemo');
  const toggle=document.getElementById('floatMemoToggle');
  let dragging=false, ox=0, oy=0, startX=0, startY=0;
  toggle.addEventListener('mousedown',e=>{
    dragging=true;
    startX=e.clientX; startY=e.clientY;
    const rect=wrap.getBoundingClientRect();
    ox=window.innerWidth-rect.right;
    oy=window.innerHeight-rect.bottom;
    e.preventDefault();
  });
  document.addEventListener('mousemove',e=>{
    if(!dragging) return;
    const dx=e.clientX-startX; const dy=e.clientY-startY;
    if(Math.abs(dx)>5||Math.abs(dy)>5){
      wrap.style.right=(ox-dx)+'px';
      wrap.style.bottom=(oy-dy)+'px';
      wrap.style.left='auto'; wrap.style.top='auto';
    }
  });
  document.addEventListener('mouseup',e=>{
    if(dragging){
      const dx=e.clientX-startX; const dy=e.clientY-startY;
      if(Math.abs(dx)<=5&&Math.abs(dy)<=5) toggleFloatMemo();
      dragging=false;
    }
  });
  /* 터치 지원 */
  toggle.addEventListener('touchstart',e=>{
    const t=e.touches[0];
    dragging=true; startX=t.clientX; startY=t.clientY;
    const rect=wrap.getBoundingClientRect();
    ox=window.innerWidth-rect.right; oy=window.innerHeight-rect.bottom;
  },{passive:true});
  document.addEventListener('touchmove',e=>{
    if(!dragging) return;
    const t=e.touches[0];
    const dx=t.clientX-startX; const dy=t.clientY-startY;
    wrap.style.right=(ox-dx)+'px';
    wrap.style.bottom=(oy-dy)+'px';
    wrap.style.left='auto'; wrap.style.top='auto';
  },{passive:true});
  document.addEventListener('touchend',()=>{ dragging=false; });
}

/* ════════════════════════════════════
   NAV 드래그 (섹션 이동 시 active)
════════════════════════════════════ */
function initDrag(){
  const navLinks=document.querySelectorAll('#topNav a[href^="#"]');
  const observer=new IntersectionObserver(entries=>{
    entries.forEach(entry=>{
      if(entry.isIntersecting){
        const id=entry.target.id;
        navLinks.forEach(a=>{
          a.classList.toggle('active', a.getAttribute('href')==='#'+id);
        });
      }
    });
  },{threshold:0.3});
  document.querySelectorAll('.sec[id]').forEach(sec=>observer.observe(sec));
}
